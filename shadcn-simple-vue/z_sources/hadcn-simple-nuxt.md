  variant?: ButtonVariants["variant"]
  size?: ButtonVariants["size"]
  class?: HTMLAttributes["class"]

  as: "button",
})
</script>

  <Primitive
    data-slot="button"
    :as="as"
    :as-child="asChild"

</template>
